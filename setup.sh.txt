#!/bin/bash

# Interactive PHPRunner Container Template Creator
# Enhanced version with user prompts and customization options

set -e

# Colors for better output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
PURPLE='\033[0;35m'
CYAN='\033[0;36m'
NC='\033[0m' # No Color

# Function to print colored output
print_color() {
    local color=$1
    local message=$2
    echo -e "${color}${message}${NC}"
}

# Function to prompt for yes/no
prompt_yn() {
    local prompt=$1
    local default=${2:-"y"}
    while true; do
        if [ "$default" = "y" ]; then
            read -p "$prompt [Y/n]: " yn
            yn=${yn:-y}
        else
            read -p "$prompt [y/N]: " yn
            yn=${yn:-n}
        fi
        case $yn in
            [Yy]* ) return 0;;
            [Nn]* ) return 1;;
            * ) echo "Please answer yes or no.";;
        esac
    done
}

# Function to prompt for input with default
prompt_input() {
    local prompt=$1
    local default=$2
    local variable_name=$3
    
    if [ -n "$default" ]; then
        read -p "$prompt [$default]: " input
        input=${input:-$default}
    else
        read -p "$prompt: " input
    fi
    
    eval "$variable_name='$input'"
}

# Main setup function
main() {
    print_color $BLUE "==================================================="
    print_color $BLUE "   Interactive PHPRunner Container Setup"
    print_color $BLUE "==================================================="
    echo

    # Container ID
    prompt_input "Enter container ID" "999" CONTAINER_ID
    
    # Check if container already exists
    if pct status $CONTAINER_ID >/dev/null 2>&1; then
        print_color $RED "Container $CONTAINER_ID already exists!"
        if prompt_yn "Do you want to destroy it and create a new one?"; then
            print_color $YELLOW "Stopping and destroying container $CONTAINER_ID..."
            pct stop $CONTAINER_ID 2>/dev/null || true
            pct destroy $CONTAINER_ID
        else
            print_color $RED "Exiting..."
            exit 1
        fi
    fi

    # Basic container settings
    prompt_input "Container hostname" "phprunner-template" HOSTNAME
    prompt_input "Memory (MB)" "1024" MEMORY
    prompt_input "Root filesystem size (GB)" "100" ROOTFS_SIZE
    
    # Storage selection
    print_color $CYAN "Available storage:"
    pvesm status | grep -E "^(local|local-lvm|local-zfs)" | awk '{print "  - " $1}'
    prompt_input "Storage name" "local" STORAGE
    
    # Network configuration
    print_color $CYAN "\nNetwork Configuration:"
    echo "1) DHCP (automatic IP)"
    echo "2) Static IP"
    prompt_input "Choose network type (1-2)" "1" NET_TYPE
    
    if [ "$NET_TYPE" = "2" ]; then
        prompt_input "Static IP (e.g., 192.168.1.100/24)" "" STATIC_IP
        prompt_input "Gateway IP" "192.168.1.1" GATEWAY
        NET_CONFIG="name=eth0,bridge=vmbr0,ip=$STATIC_IP,gw=$GATEWAY"
    else
        NET_CONFIG="name=eth0,bridge=vmbr0,ip=dhcp"
    fi

    # PHP version selection
    print_color $CYAN "\nPHP Version Selection:"
    echo "1) PHP 8.4 (Latest, recommended)"
    echo "2) PHP 8.3 (Stable)"
    echo "3) PHP 8.2 (LTS)"
    echo "4) PHP 8.1 (Legacy)"
    prompt_input "Choose PHP version (1-4)" "1" PHP_CHOICE
    
    case $PHP_CHOICE in
        1) PHP_VERSION="8.4";;
        2) PHP_VERSION="8.3";;
        3) PHP_VERSION="8.2";;
        4) PHP_VERSION="8.1";;
        *) PHP_VERSION="8.4";;
    esac

    # Additional components
    print_color $CYAN "\nAdditional Components:"
    INSTALL_REDIS=$(prompt_yn "Install Redis?" && echo "yes" || echo "no")
    INSTALL_NODEJS=$(prompt_yn "Install Node.js?" && echo "yes" || echo "no")
    INSTALL_COMPOSER=$(prompt_yn "Install Composer?" && echo "yes" || echo "no")
    INSTALL_MYSQL=$(prompt_yn "Install MySQL/MariaDB server (not just client)?" "n" && echo "yes" || echo "no")

    # SSL Configuration
    print_color $CYAN "\nSSL Configuration:"
    SETUP_SSL=$(prompt_yn "Prepare SSL infrastructure?" && echo "yes" || echo "no")
    
    if [ "$SETUP_SSL" = "yes" ]; then
        prompt_input "Default email for SSL certificates" "admin@example.com" SSL_EMAIL
    fi

    # Timezone
    print_color $CYAN "\nTimezone Configuration:"
    echo "Current system timezone: $(timedatectl show --property=Timezone --value)"
    prompt_input "PHP timezone" "America/Edmonton" TIMEZONE

    # Resource optimization
    print_color $CYAN "\nResource Optimization:"
    echo "1) Development (more resources, debug enabled)"
    echo "2) Production (optimized, debug disabled)"
    prompt_input "Choose optimization profile (1-2)" "2" OPTIMIZATION

    # Confirm settings
    print_color $YELLOW "\n=== Configuration Summary ==="
    echo "Container ID: $CONTAINER_ID"
    echo "Hostname: $HOSTNAME"
    echo "Memory: ${MEMORY}MB"
    echo "Storage: $STORAGE:${ROOTFS_SIZE}GB"
    echo "Network: $NET_CONFIG"
    echo "PHP Version: $PHP_VERSION"
    echo "Redis: $INSTALL_REDIS"
    echo "Node.js: $INSTALL_NODEJS"
    echo "Composer: $INSTALL_COMPOSER"
    echo "MySQL Server: $INSTALL_MYSQL"
    echo "SSL Ready: $SETUP_SSL"
    echo "Timezone: $TIMEZONE"
    echo "Profile: $([ "$OPTIMIZATION" = "1" ] && echo "Development" || echo "Production")"
    echo

    if ! prompt_yn "Proceed with container creation?"; then
        print_color $RED "Aborted by user."
        exit 1
    fi

    # Create container
    create_container

    # Configure container
    configure_container

    # Final steps
    finalize_setup
}

create_container() {
    print_color $GREEN "\n📦 Creating container..."
    
    pct create $CONTAINER_ID local:vztmpl/debian-12-standard_12.7-1_amd64.tar.zst \
        --hostname $HOSTNAME \
        --memory $MEMORY \
        --rootfs $STORAGE:$ROOTFS_SIZE \
        --net0 $NET_CONFIG \
        --unprivileged 1 \
        --onboot 1

    print_color $GREEN "✅ Container created successfully!"
    
    print_color $YELLOW "🚀 Starting container..."
    pct start $CONTAINER_ID
    
    # Wait for container to be ready
    print_color $YELLOW "⏳ Waiting for container to be ready..."
    sleep 10
}

configure_container() {
    print_color $GREEN "\n🔧 Configuring container..."

    # Create the configuration script
    cat > /tmp/container_config.sh << EOF
#!/bin/bash
set -e

echo '🔄 Starting container configuration...'

# Update system
echo '📦 Updating system packages...'
apt update && apt dist-upgrade -y

# Install basic packages
echo '📦 Installing basic packages...'
apt install -y apt-transport-https lsb-release ca-certificates wget curl unzip git nano htop

# Add Sury PHP repository
echo '📦 Adding Sury PHP repository...'
wget -O /etc/apt/trusted.gpg.d/php.gpg https://packages.sury.org/php/apt.gpg
echo 'deb https://packages.sury.org/php/ bookworm main' > /etc/apt/sources.list.d/php.list
apt update

# Install web server
echo '🌐 Installing nginx...'
apt install -y nginx-light

# Install PHP
echo '🐘 Installing PHP $PHP_VERSION...'
apt install -y \\
    php$PHP_VERSION-fpm \\
    php$PHP_VERSION-cli \\
    php$PHP_VERSION-mysql \\
    php$PHP_VERSION-curl \\
    php$PHP_VERSION-gd \\
    php$PHP_VERSION-zip \\
    php$PHP_VERSION-mbstring \\
    php$PHP_VERSION-xml \\
    php$PHP_VERSION-imagick \\
    php$PHP_VERSION-intl \\
    php$PHP_VERSION-opcache \\
    php$PHP_VERSION-readline

# Install database client
echo '🗄️ Installing MariaDB client...'
apt install -y mariadb-client

# Conditional installations
if [ "$INSTALL_MYSQL" = "yes" ]; then
    echo '🗄️ Installing MariaDB server...'
    apt install -y mariadb-server
    systemctl enable mariadb
    systemctl start mariadb
fi

if [ "$INSTALL_REDIS" = "yes" ]; then
    echo '🔴 Installing Redis...'
    apt install -y redis-server php$PHP_VERSION-redis
    systemctl enable redis-server
    systemctl start redis-server
fi

if [ "$INSTALL_NODEJS" = "yes" ]; then
    echo '🟢 Installing Node.js...'
    curl -fsSL https://deb.nodesource.com/setup_lts.x | bash -
    apt install -y nodejs
fi

if [ "$INSTALL_COMPOSER" = "yes" ]; then
    echo '🎼 Installing Composer...'
    curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer
fi

if [ "$SETUP_SSL" = "yes" ]; then
    echo '🔒 Installing SSL tools...'
    apt install -y certbot python3-certbot-nginx
fi

echo '✅ Package installation complete!'
EOF

    # Transfer and execute configuration
    pct push $CONTAINER_ID /tmp/container_config.sh /tmp/config.sh
    pct exec $CONTAINER_ID -- chmod +x /tmp/config.sh
    pct exec $CONTAINER_ID -- bash -c "
        export PHP_VERSION='$PHP_VERSION'
        export INSTALL_REDIS='$INSTALL_REDIS'
        export INSTALL_NODEJS='$INSTALL_NODEJS'
        export INSTALL_COMPOSER='$INSTALL_COMPOSER'
        export INSTALL_MYSQL='$INSTALL_MYSQL'
        export SETUP_SSL='$SETUP_SSL'
        /tmp/config.sh
    "

    # Configure PHP
    configure_php

    # Configure nginx
    configure_nginx

    # Create web content
    create_web_content

    # Optimize for profile
    if [ "$OPTIMIZATION" = "1" ]; then
        configure_development
    else
        configure_production
    fi

    print_color $GREEN "✅ Container configuration complete!"
}

configure_php() {
    print_color $YELLOW "🐘 Configuring PHP $PHP_VERSION..."

    # PHP-FPM optimization
    pct exec $CONTAINER_ID -- bash -c "
        if [ '$OPTIMIZATION' = '1' ]; then
            # Development settings
            sed -i 's/pm.max_children = 5/pm.max_children = 8/' /etc/php/$PHP_VERSION/fpm/pool.d/www.conf
            sed -i 's/pm.start_servers = 2/pm.start_servers = 2/' /etc/php/$PHP_VERSION/fpm/pool.d/www.conf
        else
            # Production settings
            sed -i 's/pm.max_children = 5/pm.max_children = 4/' /etc/php/$PHP_VERSION/fpm/pool.d/www.conf
            sed -i 's/pm.start_servers = 2/pm.start_servers = 1/' /etc/php/$PHP_VERSION/fpm/pool.d/www.conf
        fi
    "

    # Create PHP configuration
    pct exec $CONTAINER_ID -- bash -c "
        cat > /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini << 'EOL'
; PHPRunner optimizations for PHP $PHP_VERSION
upload_max_filesize = 50M
post_max_size = 50M
max_execution_time = 300
memory_limit = 256M
max_input_vars = 3000
date.timezone = $TIMEZONE
expose_php = Off
display_errors = $([ '$OPTIMIZATION' = '1' ] && echo 'On' || echo 'Off')
display_startup_errors = $([ '$OPTIMIZATION' = '1' ] && echo 'On' || echo 'Off')
log_errors = On
error_reporting = $([ '$OPTIMIZATION' = '1' ] && echo 'E_ALL' || echo 'E_ALL & ~E_DEPRECATED & ~E_STRICT')
EOL
    "

    # Enable and start PHP-FPM
    pct exec $CONTAINER_ID -- systemctl enable php$PHP_VERSION-fpm
    pct exec $CONTAINER_ID -- systemctl start php$PHP_VERSION-fpm
}

configure_nginx() {
    print_color $YELLOW "🌐 Configuring nginx..."

    pct exec $CONTAINER_ID -- bash -c "
        # Create web directory
        mkdir -p /var/www/html
        chown -R www-data:www-data /var/www/html

        # Configure nginx
        cat > /etc/nginx/sites-available/default << 'EOL'
server {
    listen 80 default_server;
    listen [::]:80 default_server;
    
    root /var/www/html;
    index index.php index.html index.htm;
    server_name _;
    
    # Increase upload limits for PHPRunner
    client_max_body_size 50M;
    
    # Let's Encrypt challenge location
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }
    
    location / {
        try_files \\\$uri \\\$uri/ /index.php?\\\$query_string;
    }
    
    location ~ \.php\\\$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php$PHP_VERSION-fpm.sock;
        fastcgi_param SCRIPT_FILENAME \\\$document_root\\\$fastcgi_script_name;
        include fastcgi_params;
    }
    
    location ~ /\.ht {
        deny all;
    }
    
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|pdf|zip)\\\$ {
        expires 1y;
        add_header Cache-Control 'public, immutable';
    }
}
EOL

        # Test and reload nginx
        nginx -t
        systemctl enable nginx
        systemctl start nginx
    "

    if [ "$SETUP_SSL" = "yes" ]; then
        create_ssl_script
    fi
}

create_ssl_script() {
    print_color $YELLOW "🔒 Creating SSL setup script..."

    pct exec $CONTAINER_ID -- bash -c "
        # Create SSL directories
        mkdir -p /etc/nginx/ssl
        mkdir -p /var/www/html/.well-known/acme-challenge

        # Create SSL setup script
        cat > /root/setup-ssl.sh << 'EOL'
#!/bin/bash
# SSL Setup Script for PHPRunner Container

if [ -z \\\"\\\$1\\\" ]; then
    echo \\\"Usage: ./setup-ssl.sh your-domain.com\\\"
    echo \\\"Example: ./setup-ssl.sh myapp.example.com\\\"
    exit 1
fi

DOMAIN=\\\$1
EMAIL=\\\"$SSL_EMAIL\\\"

echo \\\"Setting up SSL for domain: \\\$DOMAIN\\\"

# Update server_name in nginx config
sed -i \\\"s/server_name _;/server_name \\\$DOMAIN;/\\\" /etc/nginx/sites-available/default

# Get Let's Encrypt certificate
certbot --nginx -d \\\$DOMAIN --non-interactive --agree-tos --email \\\$EMAIL

# Enable automatic renewal
systemctl enable certbot.timer

echo \\\"SSL setup complete for \\\$DOMAIN\\\"
echo \\\"Your site is now available at: https://\\\$DOMAIN\\\"
EOL

        chmod +x /root/setup-ssl.sh
    "
}

configure_development() {
    print_color $YELLOW "🛠️ Applying development optimizations..."
    
    pct exec $CONTAINER_ID -- bash -c "
        # Enable more detailed logging
        echo 'log_level = debug' >> /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini
        
        # Install development tools
        apt install -y vim-tiny tree
        
        # Restart services to apply changes
        systemctl restart php$PHP_VERSION-fpm nginx
    "
}

configure_production() {
    print_color $YELLOW "🚀 Applying production optimizations..."
    
    pct exec $CONTAINER_ID -- bash -c "
        # Enable OPCache optimizations
        echo 'opcache.enable=1' >> /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini
        echo 'opcache.memory_consumption=64' >> /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini
        echo 'opcache.interned_strings_buffer=16' >> /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini
        echo 'opcache.max_accelerated_files=4000' >> /etc/php/$PHP_VERSION/fpm/conf.d/99-phprunner.ini
        
        # Restart services to apply changes
        systemctl restart php$PHP_VERSION-fpm nginx
    "
}

finalize_setup() {
    print_color $GREEN "\n🎉 Finalizing setup..."

    # Get container IP
    CONTAINER_IP=$(pct exec $CONTAINER_ID -- ip addr show eth0 | grep 'inet ' | awk '{print $2}' | cut -d'/' -f1 2>/dev/null || echo "DHCP")

    # Final verification
    pct exec $CONTAINER_ID -- bash -c "
        echo '🔍 Final verification:'
        echo '=================='
        echo 'PHP Version:' && php -v | head -1
        echo 'PHP-FPM Status:' && systemctl is-active php$PHP_VERSION-fpm
        echo 'Nginx Status:' && systemctl is-active nginx
        if [ '$INSTALL_MYSQL' = 'yes' ]; then
            echo 'MariaDB Status:' && systemctl is-active mariadb
        fi
        if [ '$INSTALL_REDIS' = 'yes' ]; then
            echo 'Redis Status:' && systemctl is-active redis-server
        fi
    "

    print_color $GREEN "\n✅ Container setup complete!"
    print_color $CYAN "\n📋 Container Details:"
    echo "  ID: $CONTAINER_ID"
    echo "  Hostname: $HOSTNAME"
    echo "  IP Address: $CONTAINER_IP"
    echo "  PHP Version: $PHP_VERSION"
    
    if [ "$CONTAINER_IP" != "DHCP" ]; then
        echo "  Web Access: http://$CONTAINER_IP/"
        echo "  Test Page: http://$CONTAINER_IP/test.php"
    else
        echo "  Web Access: http://[container-ip]/"
        echo "  Test Page: http://[container-ip]/test.php"
    fi

    print_color $YELLOW "\n📝 Next Steps:"
    echo "1. Test the web server: Browse to the container IP"
    echo "2. Deploy your PHPRunner application to /var/www/html/"
    if [ "$SETUP_SSL" = "yes" ]; then
        echo "3. Setup SSL: pct exec $CONTAINER_ID -- /root/setup-ssl.sh your-domain.com"
    fi
    echo "4. Convert to template: pct stop $CONTAINER_ID && pct template $CONTAINER_ID"
    
    if prompt_yn "\nWould you like to create a customer clone now?"; then
        create_customer_clone
    fi
}

create_customer_clone() {
    print_color $CYAN "\n👥 Creating customer clone..."
    
    prompt_input "Customer container ID" "101" CUSTOMER_ID
    prompt_input "Customer hostname" "customer-app" CUSTOMER_HOSTNAME
    prompt_input "Customer memory (MB)" "512" CUSTOMER_MEMORY
    prompt_input "Customer CPU cores" "1" CUSTOMER_CORES
    
    if [ "$NET_TYPE" = "2" ]; then
        prompt_input "Customer IP (e.g., 192.168.1.101/24)" "" CUSTOMER_IP
        CUSTOMER_NET="name=eth0,bridge=vmbr0,ip=$CUSTOMER_IP,gw=$GATEWAY"
    else
        CUSTOMER_NET="name=eth0,bridge=vmbr0,ip=dhcp"
    fi

    print_color $YELLOW "🔄 Cloning container..."
    pct clone $CONTAINER_ID $CUSTOMER_ID --hostname $CUSTOMER_HOSTNAME
    pct set $CUSTOMER_ID --memory $CUSTOMER_MEMORY --cores $CUSTOMER_CORES
    pct set $CUSTOMER_ID --net0 $CUSTOMER_NET
    pct start $CUSTOMER_ID

    print_color $GREEN "✅ Customer container $CUSTOMER_ID created and started!"
    
    # Get customer IP
    sleep 5
    CUSTOMER_REAL_IP=$(pct exec $CUSTOMER_ID -- ip addr show eth0 | grep 'inet ' | awk '{print $2}' | cut -d'/' -f1 2>/dev/null || echo "DHCP")
    
    print_color $CYAN "\n📋 Customer Container Details:"
    echo "  ID: $CUSTOMER_ID"
    echo "  Hostname: $CUSTOMER_HOSTNAME"
    echo "  IP Address: $CUSTOMER_REAL_IP"
    
    if [ "$CUSTOMER_REAL_IP" != "DHCP" ]; then
        echo "  Web Access: http://$CUSTOMER_REAL_IP/"
    fi
}

# Run main function
main "$@"
